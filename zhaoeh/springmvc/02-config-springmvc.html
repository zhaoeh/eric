springmvc配置 1、web.xml
        搭建springmvc工程，首先配置web.xml：
        
            <?xml version="1.0" encoding="UTF-8"?>
            <web-app xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance"
                     xmlns="http://java.sun.com/xml/ns/javaee"
                     xsi:schemaLocation="http://java.sun.com/xml/ns/javaee http://java.sun.com/xml/ns/javaee/web-app_2_5.xsd"
                     id="WebApp_ID" version="2.5">
                <display-name>myspringmvc</display-name>

                <!-- 一切web框架从web.xml开始分析 -->
                <!-- 下面的配置可以说是spring的默认配置。 ContextLoaderListener 这个监听器如果想在Servlet中获取spring的bean实例的话必须配置 -->
                <!-- 加载spring容器，将spring相关的配置文件通过通配符的方式一次性加载进web容器中 -->
                <!-- 全局初始化参数contextConfigLocation，绑定spring文件，可配置多个文件，用逗号分隔，可使用通配符* -->
                <context-param>
                    <!-- 如果不配置该参数，则ContextLoaderListener默认加载/WEB-INF/applicationContext.xml -->
                    <param-name>contextConfigLocation</param-name>
                    <param-value>classpath:/spring/applicationContext.xml</param-value>
                </context-param>
                <!-- 配置spring的监听器，用于启动spring容器，加载全局参数contextConfigLocation -->
                <listener>
                    <listener-class>org.springframework.web.context.ContextLoaderListener</listener-class>
                </listener>

                <!-- springmvc的GET请求乱码需要手动对http请求参数转码（和Servlet一样），对于POST请求使用springmvc自带的过滤器实现 -->
                <filter>
                    <filter-name>characterEncodingFilter</filter-name>
                    <filter-class>org.springframework.web.filter.CharacterEncodingFilter</filter-class>
                    <init-param>
                        <param-name>encoding</param-name>
                        <param-value>UTF-8</param-value>
                    </init-param>
                    <init-param>
                        <param-name>forceEncoding</param-name>
                        <param-value>true</param-value>
                    </init-param>
                </filter>
                <filter-mapping>
                    <!-- 配置filter过滤路径 -->
                    <filter-name>characterEncodingFilter</filter-name>
                    <url-pattern>/*</url-pattern>
                </filter-mapping>

                <!-- springmvc的restful风格实现：浏览器form只支持GET和POST，不支持 DELETE和PUT请求，Spring添加了一个过滤器filter，可以将这些请求转换成标准的http方法 -->
                <!-- 该过滤器支持转换成GET,POST,DELETE,PUT请求 -->
                <filter>
                    <filter-name>HidenHttpMethodFilter</filter-name>
                    <filter-class>org.springframework.web.filter.HiddenHttpMethodFilter</filter-class>
                </filter>
                <filter-mapping>
                    <filter-name>HidenHttpMethodFilter</filter-name>
                    <!-- 对所有的请求都进行过滤 -->
                    <url-pattern>/*</url-pattern>
                </filter-mapping>

                <!-- 配置前端控制器DispatcherServlet -->
                <servlet>
                    <servlet-name>springmvc</servlet-name>
                    <servlet-class>org.springframework.web.servlet.DispatcherServlet</servlet-class>
                    <init-param>
                        <!-- servlet初始化参数contextConfigLocation，绑定spring文件，可配置多个文件，用逗号分隔，可使用通配符* -->
                        <!-- 如果不配置该参数，则DispatcherServlet默认加载/WEB-INF/${servlet-name}-servlet.xml -->
                        <!-- 如果上面同时有全局的配置文件的话， 此处指定的springmvc配置文件不要和上面全局的一致，否则将加载两次bean -->
                        <!-- 建议：其他容器比如struts，servlet，filter等bean的配置单独配置到上面的文件中 -->
                        <param-name>contextConfigLocation</param-name>
                        <param-value>classpath:/springmvc/springmvc-base.xml</param-value>
                    </init-param>
                    <load-on-startup>1</load-on-startup>
                </servlet>
                <!-- 访问DispatcherServlet对应的路径 -->
                <servlet-mapping>
                    <servlet-name>springmvc</servlet-name>
                    <!-- /*强迫所有请求都经过目标Servlet；/*的话会拦截所有请求，包括jsp，当请求一个jsp时，依然使用dispatcherServlet解析jsp地址，最终不能找到handler报错 -->
                    <!-- springmvc默认会拦截符合指定url规则的请求，然后统一使用springmvc这一套东西进行跳转 -->
                    <!-- 使用springmvc配置了/后，则所有的请求都有springmvc进行转发，包括对静态资源的请求 -->
                    <!-- 但是，当配置为/*时，尽管你真正请求的是个jsp页面，它也会报错 -->
                    <!-- / 将使你配置的servlet成为默认的servlet。/不拦截jsp资源! -->
                    <url-pattern>/</url-pattern><!--/不过滤jsp防止死循环 -->
                </servlet-mapping>

                <!-- 自定义Servlet -->
                <servlet>
                    <servlet-name>myServlet</servlet-name>
                    <servlet-class>myspringmvc.model.demo.servlet.MyServlet</servlet-class>
                </servlet>
                <servlet-mapping>
                    <servlet-name>myServlet</servlet-name>
                    <url-pattern>/servlet/MyServlet</url-pattern>
                </servlet-mapping>
                <welcome-file-list>
                    <welcome-file>index.html</welcome-file>
                    <welcome-file>index.htm</welcome-file>
                    <welcome-file>index.jsp</welcome-file>
                    <welcome-file>default.html</welcome-file>
                    <welcome-file>default.htm</welcome-file>
                    <welcome-file>default.jsp</welcome-file>
                </welcome-file-list>
            </web-app>
2、applicationContext.xml
        applicationContext.xml是对service，dao层的spring容器的描述，被web.xml中的ContextLoaderListener负责调用。
        
            <?xml version="1.0" encoding="UTF-8"?>
            <beans xmlns="http://www.springframework.org/schema/beans"
                   xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance"
                   xmlns:context="http://www.springframework.org/schema/context"
                   xsi:schemaLocation="http://www.springframework.org/schema/beans
                http://www.springframework.org/schema/beans/spring-beans.xsd
                http://www.springframework.org/schema/context http://www.springframework.org/schema/context/spring-context.xsd">

                <!-- 扫描spring包 -->
                <context:component-scan base-package="myspringmvc.model">
                    <!-- 排除扫描springMvc的controller层注解 -->
                    <context:exclude-filter type="annotation" expression="org.springframework.stereotype.Controller"/>
                </context:component-scan>

            </beans>
3、 springmvc-base.xml
        springmvc-base.xml是对controller层的springmvc容器，被web.xml中的DispatcherServlet负责调用。
        
            <?xml version="1.0" encoding="UTF-8"?>
            <beans xmlns="http://www.springframework.org/schema/beans"
                   xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance"
                   xmlns:context="http://www.springframework.org/schema/context"
                   xmlns:mvc="http://www.springframework.org/schema/mvc"
                   xsi:schemaLocation="http://www.springframework.org/schema/beans
                http://www.springframework.org/schema/beans/spring-beans.xsd
                http://www.springframework.org/schema/context http://www.springframework.org/schema/context/spring-context.xsd
                http://www.springframework.org/schema/mvc
                http://www.springframework.org/schema/mvc/spring-mvc.xsd">

                <mvc:annotation-driven/>
                <!--注解驱动 -->

                <mvc:resources mapping="/static/**" location="/static/"></mvc:resources>

                <!-- 扫描controller；springmvc建议只扫描controller -->
                <context:component-scan base-package="myspringmvc.model.demo.controller"/>

                <!-- 视图解析器 -->
                <bean class="org.springframework.web.servlet.view.InternalResourceViewResolver">
                    <property name="prefix" value="/WEB-INF/jsp/"/><!-- jsp所在的位置 -->
                    <property name="suffix" value=".jsp"></property><!-- jsp文件的后缀名 -->
                </bean>

            </beans>
4、 helloWorld.jsp
        在WEB-INF/jsp目录下创建一个静态页面jsp，用于充当Servlet容器中的静态资源。
        
            <%-- 输出简单的helloWorld --%>
            <%@page language="java" contentType="text/html;charset=utf-8"
                pageEncoding="utf-8"%>
            <html>
            <head>
            <title>www.zhaoerhu.com</title>
            </head>
            <body>
                <%
                    request.setCharacterEncoding("utf-8");
                %>
                <%
                    //同样，在jsp中也可以使用内置的out对象向前端浏览器输出html标签和语句等，但是不建议在jsp中使用out.println()的方式；
                    //更好的应该是直接使用jsp的表达式输出语句
                    out.println("<center><h1>HelloWorld!!!</h1></center>");
                %>
                <%="<center><h1>欢迎来到Springmvc的世界！</h1></center>"%>
            </body>
            </html>
